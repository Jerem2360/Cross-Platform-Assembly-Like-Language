sym input:
    data int.32 12;
sym output:
    data int.32;
sym string:
    data int.8 "Output is %i.";



fn fibo2($input: int.32) -> int.32 {
    $i: int.32;
    $tmp: int.32;
    $previous: int.32;

    
    if $input > 2 goto body;
    result <- 2;  // fibo(2) = 2
    if $input == 2 goto end;
    result <- 1;  // fibo(1) and fibo(0) are 1
    goto end;

body:

    $i <- 2;
    result <- 2;  // current
    $previous <- 1;

loop:
    if $i >= $input goto end;
    $i++;

    $tmp <- $previous;
    $previous <- result;
    result += $tmp;

    goto loop;

end:
    return;
    
}


sym fibo:
fn {
    $input: int.32;
    $output: int.32;
    $i: int.32;
    $tmp: int.32;
    $previous: int.32;

    $input <- input[int.32];

    if $input > 2 goto body;
    $output <- 2;  // fibo(2) = 2
    if $input == 2 goto end;
    $output <- 1;  // fibo(1) and fibo(0) are 1
    goto end;

body:

    $i <- 2;
    $output <- 2;  // current
    $previous <- 1;

loop:
    if $i >= $input goto end;
    $i++;

    $tmp <- $previous;
    $previous <- $output;
    $output += $tmp;

    goto loop;

end:
    output[int.32] <- $output;
    return;
}


entry {
    $fibo_res: int.32;  // to store the output of our functions

    call fibo;  // function call, uses global storage for in/out
    $fibo_res <- output[int.32];  // pick the input from the out global storage 
    call _printf(string, $fibo_res);  // print it to the terminal

    call fibo2(12) -> $fibo_res;  // function call, using parameters this time. result is placed in $fibo_res
    call _printf(string, $fibo_res);  // print our result
    exit;
}
